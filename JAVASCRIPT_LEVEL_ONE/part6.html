If we need to stop the loop we can use operator "break"

var y = 0;
while(y < 5){
console.log("y is currently: "+y);
if (y===3){
console.log("y is equal to three");
break;
}
console.log("y i still less than 5, adding 1 to y");
y = y+1;
}
