Let's create some class:
class dog():
    species = "mammal"
    def __init__(self,breed,name):
        self.breed = breed
        self.name = name

mydog = dog(breed = "Lab",name = "Sammy")
anotherway = dog("pudel","Bob")

print(mydog.breed)
print(mydog.name)
print(anotherway.breed, anotherway.name)
print(anotherway.species)

>>> Lab
>>> Sammy
>>> pudel Bob
>>> mammal
